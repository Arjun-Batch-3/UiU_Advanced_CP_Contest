#include <bits/stdc++.h>
using namespace std;

int minW(int ind) 
{
    if (ind >= 101)
        return ind - 10;
    else
        return minW(minW(ind + 11)); 
}

int main() 
{
    int t;
    while (true)
     {
        cin >> t;
        if (t == 0) 
        break;
        cout << "f91(" << t << ") = " << minW(t) << endl;
    }
    return 0;
}
