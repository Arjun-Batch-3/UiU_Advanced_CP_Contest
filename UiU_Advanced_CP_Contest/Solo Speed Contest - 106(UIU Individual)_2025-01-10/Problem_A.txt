#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    cin >> s;
    int c1 = 0, c2 = 0;
    for (int i = 0; i < s.size(); i++)
    {
        if (s[i] >= 'A' && s[i] <= 'Z')
        {
            c1++;
        }
        else if (s[i] >= 'a' && s[i] <= 'z')
        {
            c2++;
        }
    }
    cout << c1 << " " << c2 << endl;
    return 0;
}
